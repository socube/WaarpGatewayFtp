194     private String index() {
195         String index = REQUEST.index.readFileUnique();
196         StringBuilder builder = new StringBuilder(index);
197         WaarpStringUtils.replace(builder, "XXXLOCALXXX",
198                 Integer.toString(
199                         FileBasedConfiguration.fileBasedConfiguration.
200                                 getFtpInternalConfiguration().getNumberSessions())
201                         + " " + Thread.activeCount());
202         TrafficCounter trafficCounter =
203                 FileBasedConfiguration.fileBasedConfiguration.getFtpInternalConfiguration()
204                         .getGlobalTrafficShapingHandler().trafficCounter();
205         WaarpStringUtils.replace(builder, "XXXBANDWIDTHXXX",
206                 "IN:" + (trafficCounter.lastReadThroughput() / 131072) +
207                         "Mbits&nbsp;<br>&nbsp;OUT:" +
208                         (trafficCounter.lastWriteThroughput() / 131072) + "Mbits");
209         WaarpStringUtils.replaceAll(builder, "XXXHOSTIDXXX",
210                 FileBasedConfiguration.fileBasedConfiguration.HOST_ID);
211         WaarpStringUtils.replaceAll(builder, "XXXADMINXXX",
212                 "Administrator Connected");
213         WaarpStringUtils.replace(builder, "XXXVERSIONXXX",
214                 Version.ID);
215         return builder.toString();

228     private String System() {
229         getParams();
230         FtpConstraintLimitHandler handler =
231                 FileBasedConfiguration.fileBasedConfiguration.constraintLimitHandler;
232         if (params == null) {
233             String system = REQUEST.System.readFileUnique();
234             StringBuilder builder = new StringBuilder(system);
235             WaarpStringUtils.replace(builder, "XXXXCHANNELLIMITRXXX",
236                     Long.toString(FileBasedConfiguration.fileBasedConfiguration
237                             .getServerGlobalReadLimit()));
238             WaarpStringUtils.replace(builder, "XXXXCPULXXX",
239                     Double.toString(handler.getCpuLimit()));
240             WaarpStringUtils.replace(builder, "XXXXCONLXXX",
241                     Integer.toString(handler.getChannelLimit()));
242             WaarpStringUtils.replace(builder, "XXXRESULTXXX", "");
243             return builder.toString();
244         }
245         String extraInformation = null;
246         if (params.containsKey("ACTION")) {
247             List<String> action = params.get("ACTION");
248             for (String act : action) {
249                 if (act.equalsIgnoreCase("Disconnect")) {
250                     String logon = Logon();
251                     newSession = true;
252                     clearSession();
253                     forceClose = true;
254                     return logon;
255                 } else if (act.equalsIgnoreCase("Shutdown")) {
256                     String error = error("Shutdown in progress");
257                     newSession = true;
258                     clearSession();
259                     forceClose = true;
260                     shutdown = true;
261                     return error;
262                 } else if (act.equalsIgnoreCase("Validate")) {
263                     String bglobalr = getTrimValue("BGLOBR");
264                     long lglobal = FileBasedConfiguration.fileBasedConfiguration
265                             .getServerGlobalReadLimit();
266                     if (bglobalr != null) {
267                         lglobal = Long.parseLong(bglobalr);
268                     }
269                     FileBasedConfiguration.fileBasedConfiguration.changeNetworkLimit(lglobal,
270                             lglobal);
271                     bglobalr = getTrimValue("CPUL");
272                     double dcpu = handler.getCpuLimit();
273                     if (bglobalr != null) {
274                         dcpu = Double.parseDouble(bglobalr);
275                     }
276                     handler.setCpuLimit(dcpu);
277                     bglobalr = getTrimValue("CONL");
278                     int iconn = handler.getChannelLimit();
279                     if (bglobalr != null) {
280                         iconn = Integer.parseInt(bglobalr);
281                     }
282                     handler.setChannelLimit(iconn);
283                     extraInformation = "Configuration Saved";
284                 }
285             }
286         }
287         String system = REQUEST.System.readFileUnique();
288         StringBuilder builder = new StringBuilder(system);
289         WaarpStringUtils.replace(builder, "XXXXCHANNELLIMITRXXX",
290                 Long.toString(FileBasedConfiguration.fileBasedConfiguration
291                         .getServerGlobalReadLimit()));
292         WaarpStringUtils.replace(builder, "XXXXCPULXXX",
293                 Double.toString(handler.getCpuLimit()));
294         WaarpStringUtils.replace(builder, "XXXXCONLXXX",
295                 Integer.toString(handler.getChannelLimit()));
296         if (extraInformation != null) {
297             WaarpStringUtils.replace(builder, "XXXRESULTXXX", extraInformation);
298         } else {
299             WaarpStringUtils.replace(builder, "XXXRESULTXXX", "");
300         }
301         return builder.toString();
302     }

374     private String Transfer() {
375         getParams();
376         String head = REQUEST.Transfer.readHeader();
377         String end = REQUEST.Transfer.readEnd();
378         String body = REQUEST.Transfer.readBody();
379         if (params == null || (!DbConstant.admin.isActive)) {
380             end = end.replace("XXXRESULTXXX", "");
381             body = FileBasedConfiguration.fileBasedConfiguration.getHtmlTransfer(body, LIMITROW);
382             return head + body + end;
383         }
384         String message = "";
385         List<String> parms = params.get("ACTION");
386         if (parms != null) {
387             String parm = parms.get(0);
388             boolean purgeAll = false;
389             boolean purgeCorrect = false;
390             boolean delete = false;
391             if ("PurgeCorrectTransferLogs".equalsIgnoreCase(parm)) {
392                 purgeCorrect = true;
393             } else if ("PurgeAllTransferLogs".equalsIgnoreCase(parm)) {
394                 purgeAll = true;
395             } else if ("Delete".equalsIgnoreCase(parm)) {
396                 delete = true;
397             }
398             if (purgeCorrect) {
399                 DbPreparedStatement preparedStatement = null;
400                 try {
401                     preparedStatement =
402                             DbTransferLog.getStatusPrepareStament(dbSession,
403                                     ReplyCode.REPLY_250_REQUESTED_FILE_ACTION_OKAY, 0);
404                 } catch (WaarpDatabaseNoConnectionException e) {
405                     message = "Error during purge";
406                 } catch (WaarpDatabaseSqlException e) {
407                     message = "Error during purge";
408                 }
409                 if (preparedStatement != null) {
410                     try {
411                         FileBasedConfiguration config = FileBasedConfiguration.fileBasedConfiguration;
412                         String filename =
413                                 config.getBaseDirectory() +
414                                         FtpDir.SEPARATOR + config.ADMINNAME + FtpDir.SEPARATOR +
415                                         config.HOST_ID + "_logs_" + System.currentTimeMillis()
416                                         + ".xml";
417                         message = DbTransferLog.saveDbTransferLogFile(preparedStatement, filename);
418                     } finally {
419                         preparedStatement.realClose();
420                     }
421                 }
422             } else if (purgeAll) {
423                 DbPreparedStatement preparedStatement = null;
424                 try {
425                     preparedStatement =
426                             DbTransferLog.getStatusPrepareStament(dbSession,
427                                     null, 0);
428                 } catch (WaarpDatabaseNoConnectionException e) {
429                     message = "Error during purgeAll";
430                 } catch (WaarpDatabaseSqlException e) {
431                     message = "Error during purgeAll";
432                 }
433                 if (preparedStatement != null) {
434                     try {
435                         FileBasedConfiguration config = FileBasedConfiguration.fileBasedConfiguration;
436                         String filename =
437                                 config.getBaseDirectory() +
438                                         FtpDir.SEPARATOR + config.ADMINNAME + FtpDir.SEPARATOR +
439                                         config.HOST_ID + "_logs_" + System.currentTimeMillis()
440                                         + ".xml";
441                         message = DbTransferLog.saveDbTransferLogFile(preparedStatement, filename);
442                     } finally {
443                         preparedStatement.realClose();
444                     }
445                 }
446             } else if (delete) {
447                 String user = getTrimValue("user");
448                 String acct = getTrimValue("account");
449                 String specid = getTrimValue("specialid");
450                 long specialId = Long.parseLong(specid);
451                 try {
452                     DbTransferLog log = new DbTransferLog(dbSession, user, acct, specialId);
453                     FileBasedConfiguration config = FileBasedConfiguration.fileBasedConfiguration;
454                     String filename =
455                             config.getBaseDirectory() +
456                                     FtpDir.SEPARATOR + config.ADMINNAME + FtpDir.SEPARATOR +
457                                     config.HOST_ID + "_log_" + System.currentTimeMillis() + ".xml";
458                     message = log.saveDbTransferLog(filename);
459                 } catch (WaarpDatabaseException e) {
460                     message = "Error during delete 1 Log";
461                 }
462             } else {
463                 message = "No Action";
464             }
465             end = end.replace("XXXRESULTXXX", message);
466         }
467         end = end.replace("XXXRESULTXXX", "");
468         body = FileBasedConfiguration.fileBasedConfiguration.getHtmlTransfer(body, LIMITROW);
469         return head + body + end;
470     }

472     private String User() {
473         getParams();
474         String head = REQUEST.User.readHeader();
475         String end = REQUEST.User.readEnd();
476         String body = REQUEST.User.readBody();
477         FileBasedConfiguration config = FileBasedConfiguration.fileBasedConfiguration;
478         String filedefault = config.getBaseDirectory() +
479                 FtpDir.SEPARATOR + config.ADMINNAME +
480                 FtpDir.SEPARATOR + "authentication.xml";
481         if (params == null) {
482             end = end.replace("XXXRESULTXXX", "");
483             end = end.replace("XXXFILEXXX", filedefault);
484             body = FileBasedConfiguration.fileBasedConfiguration.getHtmlAuth(body);
485             return head + body + end;
486         }
487         List<String> parms = params.get("ACTION");
488         if (parms != null) {
489             String parm = parms.get(0);
490             if ("ImportExport".equalsIgnoreCase(parm)) {
491                 String file = getTrimValue("file");
492                 String exportImport = getTrimValue("export");
493                 String message = "";
494                 boolean purge = false;
495                 purge = params.containsKey("purge");
496                 boolean replace = false;
497                 replace = params.containsKey("replace");
498                 if (file == null) {
499                     file = filedefault;
500                 }
501                 end = end.replace("XXXFILEXXX", file);
502                 if (exportImport.equalsIgnoreCase("import")) {
503                     if (!config.initializeAuthent(file, purge)) {
504                         message += "Cannot initialize Authentication from " + file;
505                     } else {
506                         message += "Initialization of Authentication OK from " + file;
507                         if (replace) {
508                             if (!config.saveAuthenticationFile(
509                                     config.authenticationFile)) {
510                                 message += " but cannot replace server authenticationFile";
511                             } else {
512                                 message += " and replacement done";
513                             }
514                         }
515                     }
516                 } else {
517                     // export
518                     if (!config.saveAuthenticationFile(file)) {
519                         message += "Authentications CANNOT be saved into " + file;
520                     } else {
521                         message += "Authentications saved into " + file;
522                     }
523                 }
524                 end = end.replace("XXXRESULTXXX", message);
525             } else {
526                 end = end.replace("XXXFILEXXX", filedefault);
527             }
528         }
529         end = end.replace("XXXRESULTXXX", "");
530         body = FileBasedConfiguration.fileBasedConfiguration.getHtmlAuth(body);
531         return head + body + end;
532     }

534     private void getParams() {
535         if (request.method() == HttpMethod.GET) {
536             params = null;
537         } else if (request.method() == HttpMethod.POST) {
538             ByteBuf content = request.content();
539             if (content.isReadable()) {
540                 String param = content.toString(WaarpStringUtils.UTF8);
541                 QueryStringDecoder queryStringDecoder2 = new QueryStringDecoder("/?" + param);
542                 params = queryStringDecoder2.parameters();
543             } else {
544                 params = null;
545             }
546         }
547     }

549     private void clearSession() {
550         if (admin != null) {
551             FileBasedAuth auth = sessions.remove(admin.value());
552             DbSession ldbsession = dbSessions.remove(admin.value());
553             admin = null;
554             if (auth != null) {
555                 auth.clear();
556             }
557             if (ldbsession != null) {
558                 ldbsession.disconnect();
559                 DbAdmin.nbHttpSession--;
560             }
561         }

564     private void checkAuthent(ChannelHandlerContext ctx) {
565         newSession = true;
566         if (request.method() == HttpMethod.GET) {
567             String logon = Logon();
568             responseContent.append(logon);
569             clearSession();
570             writeResponse(ctx);
571             return;
572         } else if (request.method() == HttpMethod.POST) {
573             getParams();
574             if (params == null) {
575                 String logon = Logon();
576                 responseContent.append(logon);
577                 clearSession();
578                 writeResponse(ctx);
579                 return;
580             }
581         }
582         boolean getMenu = false;
583         if (params.containsKey("Logon")) {
584             String name = null, password = null;
585             List<String> values = null;
586             if (!params.isEmpty()) {
587                 // get values
588                 if (params.containsKey("name")) {
589                     values = params.get("name");
590                     if (values != null) {
591                         name = values.get(0);
592                         if (name == null || name.length() == 0) {
593                             getMenu = true;
594                         }
595                     }
596                 } else {
597                     getMenu = true;
598                 }
599                 // search the nb param
600                 if ((!getMenu) && params.containsKey("passwd")) {
601                     values = params.get("passwd");
602                     if (values != null) {
603                         password = values.get(0);
604                         if (password == null || password.length() == 0) {
605                             getMenu = true;
606                         } else {
607                             getMenu = false;
608                         }
609                     } else {
610                         getMenu = true;
611                     }
612                 } else {
613                     getMenu = true;
614                 }
615             } else {
616                 getMenu = true;
617             }
618             if (!getMenu) {
619                 logger.debug("Name=" + name + " vs "
620                         + name.equals(FileBasedConfiguration.fileBasedConfiguration.ADMINNAME) +
621                         " Passwd=" + password + " vs " +
622                         FileBasedConfiguration.fileBasedConfiguration.checkPassword(password));
623                 if (name.equals(FileBasedConfiguration.fileBasedConfiguration.ADMINNAME) &&
624                         FileBasedConfiguration.fileBasedConfiguration.checkPassword(password)) {
625                     authentHttp
626                             .specialNoSessionAuth(FileBasedConfiguration.fileBasedConfiguration.HOST_ID);
627                 } else {
628                     getMenu = true;
629                 }
630                 if (!authentHttp.isIdentified()) {
631                     logger.debug("Still not authenticated: {}", authentHttp);
632                     getMenu = true;
633                 }
634                 // load DbSession
635                 if (this.dbSession == null) {
636                     try {
637                         if (DbConstant.admin.isActive) {
638                             this.dbSession = new DbSession(DbConstant.admin, false);
639                             DbAdmin.nbHttpSession++;
640                             this.isPrivateDbSession = true;
641                         }
642                     } catch (WaarpDatabaseNoConnectionException e1) {
643                         // Cannot connect so use default connection
644                         logger.warn("Use default database connection");
645                         this.dbSession = DbConstant.admin.session;
646                     }
647                 }
648             }
649         } else {
650             getMenu = true;
651         }
652         if (getMenu) {
653             String logon = Logon();
654             responseContent.append(logon);
655             clearSession();
656             writeResponse(ctx);
657         } else {
658             String index = index();
659             responseContent.append(index);
660             clearSession();
661             admin = new DefaultCookie(FTPSESSION,
662                     FileBasedConfiguration.fileBasedConfiguration.HOST_ID +
663                             Long.toHexString(random.nextLong()));
664             sessions.put(admin.value(), this.authentHttp);
665             if (this.isPrivateDbSession) {
666                 dbSessions.put(admin.value(), dbSession);
667             }
668             logger.debug("CreateSession: " + uriRequest + ":{}", admin);
669             writeResponse(ctx);
670         }

673     @Override
674     protected void channelRead0(ChannelHandlerContext ctx, FullHttpRequest msg) throws Exception {
675         this.request = msg;
676         queryStringDecoder = new QueryStringDecoder(request.uri());
677         uriRequest = queryStringDecoder.path();
678         if (uriRequest.contains("gre/") || uriRequest.contains("img/") ||
679                 uriRequest.contains("res/")) {
680             HttpWriteCacheEnable.writeFile(request,
681                     ctx,
682                     FileBasedConfiguration.fileBasedConfiguration.httpBasePath + uriRequest,
683                     FTPSESSION);
684             return;
685         }
686         checkSession(ctx.channel());
687         if (!authentHttp.isIdentified()) {
688             logger.debug("Not Authent: " + uriRequest + ":{}", authentHttp);
689             checkAuthent(ctx);
690             return;
691         }
692         String find = uriRequest;
693         if (uriRequest.charAt(0) == '/') {
694             find = uriRequest.substring(1);
695         }
696         find = find.substring(0, find.indexOf("."));
697         REQUEST req = REQUEST.index;
698         try {
699             req = REQUEST.valueOf(find);
700         } catch (IllegalArgumentException e1) {
701             req = REQUEST.index;
702             logger.debug("NotFound: " + find + ":" + uriRequest);
703         }
704         switch (req) {
705             case index:
706                 responseContent.append(index());
707                 break;
708             case Logon:
709                 responseContent.append(index());
710                 break;
711             case System:
712                 responseContent.append(System());
713                 break;
714             case Rule:
715                 responseContent.append(Rule());
716                 break;
717             case User:
718                 responseContent.append(User());
719                 break;
720             case Transfer:
721                 responseContent.append(Transfer());
722                 break;
723             default:
724                 responseContent.append(index());
725                 break;
726         }
727         writeResponse(ctx);
728     }

730     private void checkSession(Channel channel) {
731         String cookieString = request.headers().get(HttpHeaderNames.COOKIE);
732         if (cookieString != null) {
733             Set<Cookie> cookies = ServerCookieDecoder.LAX.decode(cookieString);
734             if (!cookies.isEmpty()) {
735                 for (Cookie elt : cookies) {
736                     if (elt.name().equalsIgnoreCase(FTPSESSION)) {
737                         admin = elt;
738                         break;
739                     }
740                 }
741             }
742         }
743         if (admin != null) {
744             FileBasedAuth auth = sessions.get(admin.value());
745             if (auth != null) {
746                 authentHttp = auth;
747             }
748             DbSession dbSession = dbSessions.get(admin.value());
749             if (dbSession != null) {
750                 this.dbSession = dbSession;
751             }
752         } else {
753             logger.debug("NoSession: " + uriRequest + ":{}", admin);
754         }
755     }

757     private void handleCookies(HttpResponse response) {
758         String cookieString = request.headers().get(HttpHeaderNames.COOKIE);
759         if (cookieString != null) {
760             Set<Cookie> cookies = ServerCookieDecoder.LAX.decode(cookieString);
761             if (!cookies.isEmpty()) {
762                 // Reset the sessions if necessary.
763                 boolean findSession = false;
764                 for (Cookie cookie : cookies) {
765                     if (cookie.name().equalsIgnoreCase(FTPSESSION)) {
766                         if (newSession) {
767                             findSession = false;
768                         } else {
769                             findSession = true;
770                             response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
771                         }
772                     } else {
773                         response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(cookie));
774                     }
775                 }
776                 newSession = false;
777                 if (!findSession) {
778                     if (admin != null) {
779                         response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(admin));
780                         logger.debug("AddSession: " + uriRequest + ":{}", admin);
781                     }
782                 }
783             }
784         } else if (admin != null) {
785             logger.debug("AddSession: " + uriRequest + ":{}", admin);
786             response.headers().add(HttpHeaderNames.SET_COOKIE, ServerCookieEncoder.LAX.encode(admin));
787         }
788     }

790     /**
791      * Write the response
792      * 
793      * @param ctx
794      */
795     private void writeResponse(ChannelHandlerContext ctx) {
796         // Convert the response content to a ByteBuf.
797         ByteBuf buf = Unpooled.copiedBuffer(responseContent.toString(),
798                 WaarpStringUtils.UTF8);
799         responseContent.setLength(0);
800 
801         // Decide whether to close the connection or not.
802         boolean keepAlive = HttpHeaderUtil.isKeepAlive(request);
803         boolean close = HttpHeaderValues.CLOSE.equalsIgnoreCase(request
804                 .headers().get(HttpHeaderNames.CONNECTION)) ||
805                 (!keepAlive) || forceClose;
806 
807         // Build the response object.
808         FullHttpResponse response = new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, HttpResponseStatus.OK, buf);
809         response.headers().add(HttpHeaderNames.CONTENT_LENGTH, response.content().readableBytes());
810         response.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/html");
811         if (keepAlive) {
812             response.headers().set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
813         }
814         if (!close) {
815             // There's no need to add 'Content-Length' header
816             // if this is the last response.
817             response.headers().set(HttpHeaderNames.CONTENT_LENGTH,
818                     String.valueOf(buf.readableBytes()));
819         }
820 
821         handleCookies(response);
822 
823         // Write the response.
824         ChannelFuture future = ctx.writeAndFlush(response);
825         // Close the connection after the write operation is done if necessary.
826         if (close) {
827             future.addListener(WaarpSslUtility.SSLCLOSE);
828         }
829         if (shutdown) {
830             /*
831              * Thread thread = new Thread( new FtpChannelUtils(
832              * FileBasedConfiguration.fileBasedConfiguration)); thread.setDaemon(true);
833              * thread.setName("Shutdown Thread"); thread.start();
834              */
835             FtpChannelUtils.teminateServer(FileBasedConfiguration.fileBasedConfiguration);
836             if (!close) {
837                 future.addListener(WaarpSslUtility.SSLCLOSE);
838             }
839         }
840     }

842     /**
843      * Send an error and close
844      * 
845      * @param ctx
846      * @param status
847      */
848     private void sendError(ChannelHandlerContext ctx, HttpResponseStatus status) {
849         responseContent.setLength(0);
850         responseContent.append(error(status.toString()));
851         FullHttpResponse response = new DefaultFullHttpResponse(
852                 HttpVersion.HTTP_1_1, status, Unpooled.copiedBuffer(responseContent.toString(), WaarpStringUtils.UTF8));
853         response.headers().add(HttpHeaderNames.CONTENT_LENGTH, response.content().readableBytes());
854         response.headers().set(HttpHeaderNames.CONTENT_TYPE, "text/html");
855         clearSession();
856         // Close the connection as soon as the error message is sent.
857         ctx.channel().writeAndFlush(response).addListener(WaarpSslUtility.SSLCLOSE);
858     }

860     @Override
861     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
862         Throwable e1 = cause;
863         if (!(e1 instanceof CommandAbstractException)) {
864             if (e1 instanceof IOException) {
865                 // Nothing to do
866                 return;
867             }
868             logger.warn("Exception in HttpSslHandler", e1);
869         }
870         if (ctx.channel().isActive()) {
871             sendError(ctx, HttpResponseStatus.BAD_REQUEST);
872         }
873     }
